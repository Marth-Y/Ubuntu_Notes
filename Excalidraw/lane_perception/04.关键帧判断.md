bool KeyFrameManager::Judge(const int frame_id, const Odometry& curr_odom, const bool force_key_frame)
{
    bool is_key_frame = force_key_frame ||
                        key_frame_ids_.size()<key_frame_ids_cnt_thres_ ||
                        IsMoved(last_key_frame_odom_, curr_odom);
    bool is_odom_valid = (curr_odom.speed>=-1e-10 && curr_odom.speed<=60) && 
                         (std::fabs(curr_odom.yaw_rate)<=0.5);
    if(is_key_frame && is_odom_valid){
        last_key_frame_odom_ = curr_odom;
        key_frame_ids_.insert(frame_id);
        while(key_frame_ids_.size()>key_frame_ids_cnt_thres_){
            key_frame_ids_.erase(key_frame_ids_.begin());
        }
        return true;
    }
    return false;
} ^PbUVfDpf
